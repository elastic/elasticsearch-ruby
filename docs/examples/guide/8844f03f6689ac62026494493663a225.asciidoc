# Passing in the base64 encoding of id and api_key joined by a colon:
Elasticsearch::Client.new(
  host: host,
  api_key: 'BASE64_ENCODED_API_KEY'
)

# Passing in a hash with the id and api_key:
Elasticsearch::Client.new(
  host: host,
  api_key: {id: 'ID', api_key: 'API_KEY'}
)
